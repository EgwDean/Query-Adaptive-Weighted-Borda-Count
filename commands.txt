# ============================================================================
# Query-Adaptive Weighted Borda Count -- setup & run (Linux / HPC)
# ============================================================================

# 1) Clone the repository
git clone https://github.com/EgwDean/Query-Adaptive-Weighted-Borda-Count.git
cd Query-Adaptive-Weighted-Borda-Count

# 2) Create and activate a virtual environment
#    (Ubuntu 24.04 is PEP-668 "externally managed" -- a venv is required; do not
#    pip-install into the system python. If venv creation fails, run once:
#    sudo apt install -y python3.12-venv )
python3 -m venv .venv
source .venv/bin/activate
python -m pip install --upgrade pip

# 3) Install PyTorch FIRST. The rtx4090 box has driver 595 (CUDA 13.2 capable),
#    which is forward-compatible with the cu121 wheels -> use cu121.
#    CPU-only fallback: pip install torch==2.3.1
pip install torch==2.3.1 --index-url https://download.pytorch.org/whl/cu121

# 4) Install the remaining requirements
pip install -r requirements.txt

# 5) Run the Phase-1 pipeline -- ONCE PER DATASET.
#    Edit `dataset:` in config.yaml (and `split:` if needed), then:
python src/download.py            # downloads the BEIR dataset (tqdm progress)
python src/embed.py               # embeds docs + queries with all-mpnet-base-v2
python src/alpha_distribution.py  # BM25 + dense + oracle alpha + boxplots

# Repeat step 5 for each dataset you want to compare. After several datasets,
# data/results/combined_alpha_boxplot.png and data/results/alpha_summary.csv
# rank them by alpha spread / closeness to 0.5 -> pick your test bed.

# Tip (long jobs): the rtx4090 box has no scheduler and is shared, so run long
# embedding/indexing jobs inside tmux so they survive an SSH disconnect:
#   tmux new -s session        # start session;  re-attach later with: tmux attach -t wbc
#   source .venv/bin/activate && python src/embed.py
#   (detach with Ctrl-b then d)
